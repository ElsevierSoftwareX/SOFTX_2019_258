# Count the number of classes
PREFIX owl: <http://www.w3.org/2002/07/owl#>
SELECT (COUNT(DISTINCT ?class) AS ?num_classes) where {
	?class a owl:Class .
}

# Count the number of object property
PREFIX owl: <http://www.w3.org/2002/07/owl#>
SELECT (COUNT(DISTINCT ?object_property) AS ?num_obj_property) where {
	?object_property a owl:ObjectProperty .
}

# Count the number of data property
PREFIX owl: <http://www.w3.org/2002/07/owl#>
SELECT (COUNT(DISTINCT ?data_property) AS ?num_data_property) where {
	?data_property a owl:DatatypeProperty .
}

# Count number of blank nodes
SELECT (count(distinct ?x) as ?num_of_blanks) where {
	?x ?y ?z
	FILTER(isBlank(?x))
}
